Twinkle.js

features

  • transition
    • fade, crossfade, slide, swap
    • default: fade
  • interval
    • Amount of time in between each "twinkle"
    • default: 2000
  • transitionSpeed
    • Speed of "twinkle"
    • default: 1000
  • easingMethod
    • Easing method for "twinkle"
    • methods
    • default: 'easeOutQuint'
  • hidden
    • An array of image sources
    • These are the image hidden by default
  • direction
    • Direction for slide and swap transitions.
    • If no direction is provided, twinkle.js will choose alternate between random directions
    • default: none

example


$('#container').twinkle({
  transition: 'slide',
  interval: 3000,
  transitionSpeed: 1200,
  easeingMethod: 'easeOutQuint',
  direction: 'left',
  hidden: [
    'path_to_image.jpg',
    'path_to_image.jpg',
    'path_to_image.jpg',
    'path_to_image.jpg'
  ]
});

Fade


$('.fade').twinkle({
  hidden: [ARRAY]
});

Crossfade


$('.crossfade').twinkle({
  hidden: [ARRAY],
  transition: 'crossfade'
});

Slide


$('.slide').twinkle({
  hidden: [ARRAY],
  transition: 'slide',
  transitionSpeed: 1400
});

Swap


$('.swap').twinkle({
  hidden: [ARRAY],
  transition: 'swap'
});
